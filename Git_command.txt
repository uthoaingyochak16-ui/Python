### Initialize your git

git config --global user.name 'User name'
git config --global user.email 'your-email@example.com'

git Init
git add.
git commit -m"First commmit'
git remote add origin https://github.com
git branch -M main
git push -u origin main

git add.

-----------------------------------------------------------------------------------------------------------

git pull origin main
